int dp_solve(int *a, int n, int aim){
	vector&lt;vector&lt;int&gt;&gt; dp(n, vector&lt;int&gt;(aim+1, 0));

	for(int j = 1; j &lt;= aim; j++){
		dp[0][j] = INT_MAX;
