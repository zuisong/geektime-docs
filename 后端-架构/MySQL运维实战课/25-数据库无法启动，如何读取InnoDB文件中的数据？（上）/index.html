def to_datetime(value):
    second = value & 0b111111
    minute = (value >> 6) & 0b111111
    hour = (value >> (6+6)) & 0b11111
    day = (value >> (5+6+6)) & 0b11111
    year_month = (value & ((1 << 39) - 1)) >> 22
    month = year_month % 13
    year = year_month // 13
    return year, month, day, hour, minute, second
