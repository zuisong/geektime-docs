fn split(input: TokenStream) -> (Ident, TokenStream) {

    // ‰∏ÄÁõ¥ÂæÄÂêéÊâæÔºåÊâæÂà∞ struct ÂÅú‰∏ãÊù•
    while let Some(item) = input.pop_front() {
        if let TokenTree::Ident(v) = item {

    // struct ÂêéÈù¢ÔºåÂ∫îËØ•ÊòØ struct name
