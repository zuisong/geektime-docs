        return s // avoid allocation
    }

        return s // avoid allocation
    } else if n < 0 || m < n {
        n = m
    }

    var b Builder

        b.WriteString(new)
        start = j + len(old)
    }
    b.WriteString(s[start:])
    return b.String()
}
